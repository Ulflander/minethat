1   package com.ulflander.minethat.model;
2   
3   import com.google.gson.annotations.Expose;
4   
5   import java.util.ArrayList;
6   
7   /**
8    * Model of sentence. A sentence is contained into paragraphs, and contains
9    * some tokens.
10   *
11   * @since 2/24/14
12   * @author Ulflander <xlaumonier@gmail.com>
13   */
14  public class Sentence extends Text {
15  
16      /**
17       * Previous sentence in paragraph.
18       */
19      private Sentence previous;
20  
21      /**
22       * Next sentence in paragraph.
23       */
24      private Sentence next;
25  
26      /**
27       * List of tokens contained in the sentence.
28       */
29      @Expose
30      private ArrayList<Token> tokens = new ArrayList<Token>();
31  
32      /**
33       * Create a new sentence with an empty content.
34       */
35      public Sentence() {
36          super("");
37      }
38  
39      /**
40       * Create a new sentence with given content.
41       *
42       * @param raw Initial content for sentence
43       */
44      public Sentence(final String raw) {
45          super(raw);
46      }
47  
48      /**
49       * Get previous sentence in paragraph.
50       *
51       * @return Previous sentence if exist, null otherwise
52       */
53      public final Sentence getPrevious() {
54          return previous;
55      }
56  
57      /**
58       * Set previous sentence in paragraph.
59       *
60       * @param p Previous sentence in paragraph
61       */
62      public final void setPrevious(final Sentence p) {
63          this.previous = p;
64      }
65  
66      /**
67       * Get next sentence in paragraph.
68       *
69       * @return Next sentence if exist, null otherwise
70       */
71      public final Sentence getNext() {
72          return next;
73      }
74  
75  
76      /**
77       * Next previous sentence in paragraph.
78       *
79       * @param n Next sentence in paragraph
80       */
81      public final void setNext(final Sentence n) {
82          this.next = n;
83      }
84  
85      /**
86       * Get the list of tokens contained into the sentence.
87       *
88       * @return Tokens contained in sentence.
89       */
90      public final ArrayList<Token> getTokens() {
91          return tokens;
92      }
93  
94      /**
95       * Append a token to the sentence.
96       *
97       * @param token Token to append to sentence
98       */
99      public final void appendToken(final Token token) {
100         tokens.add(token);
101     }
102 
103     /**
104      * Append a string to the sentence - will be converted to token.
105      *
106      * @param str String to append to sentence.
107      */
108     public final void appendToken(final String str) {
109         appendToken(new Token(str));
110     }
111 
112     /**
113      * Get token at given index.
114      *
115      * @param pos Index of token
116      * @return Token if exist, null otherwise
117      */
118     public final Token getTokenAt(final int pos) {
119         if (tokens != null && tokens.size() > pos) {
120             return tokens.get(pos);
121         }
122         return null;
123     }
124 
125     /**
126      * Get number of tokens in sentence.
127      *
128      * @return Number of tokens in sentence
129      */
130     public final int getTokensSize() {
131         if (tokens == null) {
132             return 0;
133         }
134         return tokens.size();
135     }
136 }
