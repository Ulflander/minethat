22      public static Job fromDocument(Document d, List<String> procs) {
23          Job j = new Job();
24  
25          ArrayList<Document> documents = new ArrayList<Document>();
26          documents.add(d);
27          j.setDocuments(documents);
28  
29          ArrayList<JobProcessor> processors = new ArrayList<JobProcessor>();
30          for (String p : procs) {
31              JobProcessor jp = new JobProcessor(p);
32              processors.add(jp);
33          }
34          j.setProcessors(processors);
35  
36          return j;
37      }
