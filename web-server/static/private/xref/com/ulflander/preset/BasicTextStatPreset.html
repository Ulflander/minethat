1   package com.ulflander.preset;

3   import com.ulflander.job.JobProcessor;
4   
5   import java.util.ArrayList;
6   
7   /**
8    * A preset that only split document and returns some basic statistics about
9    * the document.
10   *
11   * @see com.ulflander.processors.augment.BasicTextStat
12   *
13   * @author Ulflander <xlaumonier@gmail.com>
14   * @since 2/23/14
15   */
16  public final class BasicTextStatPreset implements IPreset {
17  
18      /**
19       * Returns processors needed for extracting some statistics on text.
20       *
21       * @return An array of job processors
22       */
23      @Override
24      public ArrayList<JobProcessor> getProcessors() {
25          ArrayList<JobProcessor> procs = new ArrayList<JobProcessor>();
26  
27          procs.add(new JobProcessor("DocumentSplitter"));
28          procs.add(new JobProcessor("LanguageDetector"));
29          procs.add(new JobProcessor("TokenConsolidation"));
30          procs.add(new JobProcessor("TokenRecognizer"));
31          procs.add(new JobProcessor("BasicTextStat"));
32  
33          return procs;
34      }
35  }
