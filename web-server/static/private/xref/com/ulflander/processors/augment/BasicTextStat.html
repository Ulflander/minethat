1   package com.ulflander.processors.augment;
2   
3   import com.ulflander.minethat.model.Chapter;
4   import com.ulflander.minethat.model.Document;
5   import com.ulflander.minethat.model.Paragraph;
6   import com.ulflander.processors.Processor;
7   import com.ulflander.processors.ProcessorDepthControl;
8   import com.ulflander.processors.Requires;
9   
10  /**
11   * Processor that returns a staistic analysis about the document.
12   *
13   * @author Ulflander <xlaumonier@gmail.com>
14   * @since 2/21/14
15   */
16  @Requires(processors = {
17      "extract.DocumentSplitter",
18      "extract.LanguageDetector"
19  })
20  public class BasicTextStat extends Processor {
21  
22  
23      /**
24       * Initialize the processor.
25       */
26      @Override
27      public final void init() {
28          setDepthControl(ProcessorDepthControl.DOCUMENT);
29          setInitialized(true);
30      }
31  
32      /**
33       * Provides a description of this Processor.
34       *
35       * @return Description of this processor
36       */
37      @Override
38      public final String describe() {
39          return "Gives some basic statistics about the text.";
40      }
41  
42      @Override
43      public final void extractDocument(final Document doc) {
44          BasicTextStatResult res = new BasicTextStatResult();
45  
46          res.setTextLength(doc.getTextLength());
47          res.setTotalTokens(doc.getTotalToken());
48          int totalParagraph = 0;
49          int totalSentence = 0;
50  
51          for (Chapter c : doc.getChapters()) {
52              for (Paragraph p : c.getParagraphs()) {
53                  totalParagraph++;
54                  totalSentence += p.getSentencesSize();
55              }
56          }
57  
58          res.setAveSentPerParagraph(totalSentence / totalParagraph);
59          res.setAveTokPerSentence(doc.getTotalToken() / totalSentence);
60          res.setTotalParagraphs(totalParagraph);
61          res.setTotalSentences(totalSentence);
62  
63          doc.getResults().add(res);
64      }
65  }
