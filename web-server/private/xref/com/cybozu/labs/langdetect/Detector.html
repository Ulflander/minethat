13  /**
14   * {@link Detector} class is to detect language from specified text.
15   * Its instance is able to be constructed via the factory class {@link DetectorFactory}.
16   * <p/>
17   * After appending a target text to the {@link Detector} instance with {@link #append(Reader)} or {@link #append(String)},
18   * the detector provides the language detection results for target text via {@link #detect()} or {@link #getProbabilities()}.
19   * {@link #detect()} method returns a single language name which has the highest probability.
20   * {@link #getProbabilities()} methods returns a list of multiple languages and their probabilities.
21   * <p/>
22   * The detector has some parameters for language detection.
23   * See {@link #setAlpha(double)}, {@link #setMaxTextLength(int)} and {@link #setPriorMap(HashMap)}.
24   * <p/>
25   * <pre>
26   * import java.util.ArrayList;
27   * import com.cybozu.labs.com.cybozu.labs.langdetect.Detector;
28   * import com.cybozu.labs.com.cybozu.labs.langdetect.DetectorFactory;
29   * import com.cybozu.labs.com.cybozu.labs.langdetect.Language;
30   *
31   * class LangDetectSample {
32   *     public void init(String profileDirectory) throws LangDetectException {
33   *         DetectorFactory.loadProfile(profileDirectory);
34   *     }
35   *     public String detect(String text) throws LangDetectException {
36   *         Detector detector = DetectorFactory.create();
37   *         detector.append(text);
38   *         return detector.detect();
39   *     }
40   *     public ArrayList<Language> detectLangs(String text) throws LangDetectException {
41   *         Detector detector = DetectorFactory.create();
42   *         detector.append(text);
43   *         return detector.getProbabilities();
44   *     }
45   * }
46   * </pre>
47   * <p/>
48   * <ul>
49   * <li>4x faster improvement based on Elmer Garduno's code. Thanks!</li>
50   * </ul>
51   *
52   * @author Nakatani Shuyo
53   * @see DetectorFactory
54   */
55  public class Detector {
56      private static final double ALPHA_DEFAULT = 0.5;
57      private static final double ALPHA_WIDTH = 0.05;
58  
59      private static final int ITERATION_LIMIT = 1000;
60      private static final double PROB_THRESHOLD = 0.1;
61      private static final double CONV_THRESHOLD = 0.99999;
62      private static final int BASE_FREQ = 10000;
63      private static final String UNKNOWN_LANG = "unknown";
64  
65      private static final Pattern URL_REGEX = Pattern.compile("https?://[-_.?&~;+=/#0-9A-Za-z]+");
66      private static final Pattern MAIL_REGEX = Pattern.compile("[-_.0-9A-Za-z]+@[-_0-9A-Za-z]+[-_.0-9A-Za-z]+");
67  
68      private final HashMap<String, double[]> wordLangProbMap;
69      private final ArrayList<String> langlist;
70  
71      private StringBuffer text;
72      private double[] langprob = null;
73  
74      private double alpha = ALPHA_DEFAULT;
75      private int n_trial = 7;
76      private int max_text_length = 10000;
77      private double[] priorMap = null;
78      private boolean verbose = false;
79      private Long seed = null;

111     /**
112      * Set prior information about language probabilities.
113      *
114      * @param priorMap the priorMap to set
115      * @throws LangDetectException
116      */
117     public void setPriorMap(HashMap<String, Double> priorMap) throws LangDetectException {
118         this.priorMap = new double[langlist.size()];
119         double sump = 0;
120         for (int i = 0; i < this.priorMap.length; ++i) {
121             String lang = langlist.get(i);
122             if (priorMap.containsKey(lang)) {
123                 double p = priorMap.get(lang);
124                 if (p < 0)
125                     throw new LangDetectException(ErrorCode.InitParamError, "Prior probability must be non-negative.");
126                 this.priorMap[i] = p;
127                 sump += p;
128             }
129         }
130         if (sump <= 0)
131             throw new LangDetectException(ErrorCode.InitParamError, "More one of prior probability must be non-zero.");
132         for (int i = 0; i < this.priorMap.length; ++i) this.priorMap[i] /= sump;
133     }

142         this.max_text_length = max_text_length;

146     /**
147      * Append the target text for language detection.
148      * This method read the text from specified input reader.
149      * If the total size of target text exceeds the limit size specified by {@link Detector#setMaxTextLength(int)},
150      * the rest is cut down.
151      *
152      * @param reader the input reader (BufferedReader as usual)
153      * @throws IOException Can't read the reader.
154      */
155     public void append(Reader reader) throws IOException {
156         char[] buf = new char[max_text_length / 2];
157         while (text.length() < max_text_length && reader.ready()) {
158             int length = reader.read(buf);
159             append(new String(buf, 0, length));
160         }
161     }
162 
163     /**
164      * Append the target text for language detection.
165      * If the total size of target text exceeds the limit size specified by {@link Detector#setMaxTextLength(int)},
166      * the rest is cut down.
167      *
168      * @param text the target text to append
169      */
170     public void append(String text) {
171         text = URL_REGEX.matcher(text).replaceAll(" ");
172         text = MAIL_REGEX.matcher(text).replaceAll(" ");
173         char pre = 0;
174         for (int i = 0; i < text.length() && i < max_text_length; ++i) {
175             char c = NGram.normalize(text.charAt(i));
176             if (c != ' ' || pre != ' ') this.text.append(c);
177             pre = c;
178         }
179     }
180 
181     /**
182      * Cleaning text to detect
183      * (eliminate URL, e-mail address and Latin sentence if it is not written in Latin alphabet)
184      */
185     private void cleaningText() {
186         int latinCount = 0, nonLatinCount = 0;
187         for (int i = 0; i < text.length(); ++i) {
188             char c = text.charAt(i);
189             if (c <= 'z' && c >= 'A') {
190                 ++latinCount;
191             } else if (c >= '\u0300' && Character.UnicodeBlock.of(c) != Character.UnicodeBlock.LATIN_EXTENDED_ADDITIONAL) {
192                 ++nonLatinCount;
193             }
194         }
195         if (latinCount * 2 < nonLatinCount) {
196             StringBuffer textWithoutLatin = new StringBuffer();
197             for (int i = 0; i < text.length(); ++i) {
198                 char c = text.charAt(i);
199                 if (c > 'z' || c < 'A') textWithoutLatin.append(c);
200             }
201             text = textWithoutLatin;
202         }
203 
204     }

206     /**
207      * Detect language of the target text and return the language name which has the highest probability.
208      *
209      * @return detected language name which has most probability.
210      * @throws LangDetectException code = ErrorCode.CantDetectError : Can't detect because of no valid features in text
211      */
212     public String detect() throws LangDetectException {
213         ArrayList<Language> probabilities = getProbabilities();
214         if (probabilities.size() > 0) return probabilities.get(0).lang;
215         return UNKNOWN_LANG;
216     }
217 
218     /**
219      * Get language candidates which have high probabilities
220      *
221      * @return possible languages list (whose probabilities are over PROB_THRESHOLD, ordered by probabilities descendently
222      * @throws LangDetectException code = ErrorCode.CantDetectError : Can't detect because of no valid features in text
223      */
224     public ArrayList<Language> getProbabilities() throws LangDetectException {
225         if (langprob == null) detectBlock();
226 
227         ArrayList<Language> list = sortProbability(langprob);
228         return list;
229     }
230 
231     /**
232      * @throws LangDetectException
233      */
234     private void detectBlock() throws LangDetectException {
235         cleaningText();
236         ArrayList<String> ngrams = extractNGrams();
237         if (ngrams.size() == 0)
238             throw new LangDetectException(ErrorCode.CantDetectError, "no features in text");
239 
240         langprob = new double[langlist.size()];
241 
242         Random rand = new Random();
243         if (seed != null) rand.setSeed(seed);
244         for (int t = 0; t < n_trial; ++t) {
245             double[] prob = initProbability();
246             double alpha = this.alpha + rand.nextGaussian() * ALPHA_WIDTH;
247 
248             for (int i = 0; ; ++i) {
249                 int r = rand.nextInt(ngrams.size());
250                 updateLangProb(prob, ngrams.get(r), alpha);
251                 if (i % 5 == 0) {
252                     if (normalizeProb(prob) > CONV_THRESHOLD || i >= ITERATION_LIMIT)
253                         break;
254                     if (verbose)
255                         System.out.println("> " + sortProbability(prob));
256                 }
257             }
258             for (int j = 0; j < langprob.length; ++j)
259                 langprob[j] += prob[j] / n_trial;
260             if (verbose) System.out.println("==> " + sortProbability(prob));
261         }
262     }

264     /**
265      * Initialize the map of language probabilities.
266      * If there is the specified prior map, use it as initial map.
267      *
268      * @return initialized map of language probabilities
269      */
270     private double[] initProbability() {
271         double[] prob = new double[langlist.size()];
272         if (priorMap != null) {
273             for (int i = 0; i < prob.length; ++i) prob[i] = priorMap[i];
274         } else {
275             for (int i = 0; i < prob.length; ++i)
276                 prob[i] = 1.0 / langlist.size();
277         }
278         return prob;
279     }

281     /**
282      * Extract n-grams from target text
283      *
284      * @return n-grams list
285      */
286     private ArrayList<String> extractNGrams() {
287         ArrayList<String> list = new ArrayList<String>();
288         NGram ngram = new NGram();
289         for (int i = 0; i < text.length(); ++i) {
290             ngram.addChar(text.charAt(i));
291             for (int n = 1; n <= NGram.N_GRAM; ++n) {
292                 String w = ngram.get(n);
293                 if (w != null && wordLangProbMap.containsKey(w)) list.add(w);
294             }
295         }
296         return list;
297     }
298 
299     /**
300      * update language probabilities with N-gram string(N=1,2,3)
301      *
302      * @param word N-gram string
303      */
304     private boolean updateLangProb(double[] prob, String word, double alpha) {
305         if (word == null || !wordLangProbMap.containsKey(word)) return false;
306 
307         double[] langProbMap = wordLangProbMap.get(word);
308         if (verbose)
309             System.out.println(word + "(" + unicodeEncode(word) + "):" + wordProbToString(langProbMap));
310 
311         double weight = alpha / BASE_FREQ;
312         for (int i = 0; i < prob.length; ++i) {
313             prob[i] *= weight + langProbMap[i];
314         }
315         return true;
316     }

318     private String wordProbToString(double[] prob) {
319         Formatter formatter = new Formatter();
320         for (int j = 0; j < prob.length; ++j) {
321             double p = prob[j];
322             if (p >= 0.00001) {
323                 formatter.format(" %s:%.5f", langlist.get(j), p);
324             }
325         }
326         return formatter.toString();
327     }

329     /**
330      * normalize probabilities and check convergence by the maximun probability
331      *
332      * @return maximum of probabilities
333      */
334     static private double normalizeProb(double[] prob) {
335         double maxp = 0, sump = 0;
336         for (int i = 0; i < prob.length; ++i) sump += prob[i];
337         for (int i = 0; i < prob.length; ++i) {
338             double p = prob[i] / sump;
339             if (maxp < p) maxp = p;
340             prob[i] = p;
341         }
342         return maxp;
343     }
344 
345     /**
346      * @param probabilities HashMap
347      * @return lanugage candidates order by probabilities descendently
348      */
349     private ArrayList<Language> sortProbability(double[] prob) {
350         ArrayList<Language> list = new ArrayList<Language>();
351         for (int j = 0; j < prob.length; ++j) {
352             double p = prob[j];
353             if (p > PROB_THRESHOLD) {
354                 for (int i = 0; i <= list.size(); ++i) {
355                     if (i == list.size() || list.get(i).prob < p) {
356                         list.add(i, new Language(langlist.get(j), p));
357                         break;
358                     }
359                 }
360             }
361         }
362         return list;
363     }

365     /**
366      * unicode encoding (for verbose mode)
367      *
368      * @param word
369      * @return
370      */
371     static private String unicodeEncode(String word) {
372         StringBuffer buf = new StringBuffer();
373         for (int i = 0; i < word.length(); ++i) {
374             char ch = word.charAt(i);
375             if (ch >= '\u0080') {
376                 String st = Integer.toHexString(0x10000 + (int) ch);
377                 while (st.length() < 4) st = "0" + st;
378                 buf.append("\\u").append(st.subSequence(1, 5));
379             } else {
380                 buf.append(ch);
381             }
382         }
383         return buf.toString();
384     }
