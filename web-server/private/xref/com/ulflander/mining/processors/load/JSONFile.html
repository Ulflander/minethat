1   package com.ulflander.mining.processors.load;
2   
3   import com.google.gson.Gson;
4   import com.google.gson.GsonBuilder;
5   import com.ulflander.app.model.Document;
6   import com.ulflander.mining.processors.LoadProcessor;
7   import com.ulflander.mining.processors.ProcessorDepthControl;
8   import com.ulflander.utils.UlfFileUtils;
9   
10  /**
11   * Processor that saves analysis as JSON.
12   *
13   * @author Ulflander <xlaumonier@gmail.com>
14   * @since 2/21/14
15   */
16  public class JSONFile extends LoadProcessor {
17  
18      /**
19       * Initialize the processor.
20       */
21      @Override
22      public final void init() {
23          setDepthControl(ProcessorDepthControl.DOCUMENT);
24          setInitialized(true);
25      }
26  
27      /**
28       * Provides a description of this Processor.
29       *
30       * @return Description of this processor
31       */
32      @Override
33      public final String describe() {
34          return "Save media and media analysis as JSON file";
35      }
36  
37      /**
38       * Save document and analysis result as a JSON on filesystem.
39       *
40       * @param doc Document to run processor on
41       */
42      @Override
43      public final void extractDocument(final Document doc) {
44  
45          Gson gson = new GsonBuilder()
46              .setPrettyPrinting()
47              .excludeFieldsWithoutExposeAnnotation()
48              .create();
49  
50          UlfFileUtils.write("/tmp/test.json", gson.toJson(doc));
51      }
52  }
