32       *

38      }

50      /**
51       * Run processor on a Sentence.
52       *
53       * @param sentence Sentence to run processor on
54       */
55      @Override
56      public final void extractSentence(final Sentence sentence) {
57          Token token = sentence.getTokenAt(0);
58          Token next;
59          TokenType tt;
60          HashMap<TokenType, Integer> scores;
61  
62          while (token != null) {
63              next = token.getNext();
64              if (next != null) {
65                  next.consolidate(token.getScores(), 1);
66                  token.consolidate(next.getScores(), 1);
67              }
68              token = next;
69          }
70      }
71  
72      /**
73       * Consolidate token type.
74       *
75       * @param token Token to consolidate
76       * @return Token reference
77       */
78      public final Token consolidate(final Token token) {
79          return token;
80      }
81  }
