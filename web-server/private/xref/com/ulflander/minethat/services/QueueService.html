13  public class QueueService extends Service {
14  
15      @Override
16      public final void run() {
17          //  Prepare our context and sockets
18          Context context = ZMQ.context(1);
19  
20          //  Socket facing clients
21          Socket frontend = context.socket(ZMQ.ROUTER);
22          frontend.bind("tcp://127.0.0.1:5559");
23  
24          //  Socket facing services
25          Socket backend = context.socket(ZMQ.DEALER);
26          backend.bind("tcp://127.0.0.1:5560");
27  
28          //  Start the proxy
29          ZMQ.proxy(frontend, backend, null);
30  
31          //  We never get here but clean up anyhow
32          frontend.close();
33          backend.close();
34          context.term();
35      }
36  
37  }
