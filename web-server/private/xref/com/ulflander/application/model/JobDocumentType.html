1   package com.ulflander.application.model;
2   
3   /**
4    * Job document type - defines what value of JobDocument is: a file, an URL,
5    * a raw string...
6    *
7    * @author Ulflander <xlaumonier@gmail.com>
8    * @since 2/27/14
9    */
10  public enum JobDocumentType {
11  
12      /**
13       * JobDocument value is not known (will trigger an exception in extraction).
14       */
15      UNKNOWN("unknown"),
16  
17      /**
18       * JobDocument value is an URL.
19       */
20      URL("url"),
21  
22      /**
23       * JobDocument value is a path to a file.
24       */
25      FILE("file"),
26  
27      /**
28       * JobDocument value is an ID of a Document in database.
29       */
30      DOCUMENT("doc"),
31  
32      /**
33       * JobDocument value is a raw text.
34       */
35      STRING("string"),
36  
37      /**
38       * JobDocument value is an HTML string.
39       */
40      HTML_STRING("html_string");
41  
42  
43      /**
44       * Tag.
45       */
46      private final String type;
47  
48      /**
49       * Initialize job document type.
50       *
51       * @param t Tag.
52       */
53      private JobDocumentType(final String t) {
54          this.type = t;
55      }
56  
57      /**
58       * Returns the encoding for this JobDocumentType.
59       *
60       * @return A string representing the kind of job target
61       */
62      public String toString() {
63          return getType();
64      }
65  
66      /**
67       * Get type.
68       *
69       * @return Tag as string.
70       */
71      protected String getType() {
72          return this.type;
73      }
74  
75      /**
76       * Get JobDocumentType value given its string representation.
77       *
78       * @param value String representation of a part-of-speech (".", "MD"...)
79       * @return PennPOSTag representation
80       */
81      public static JobDocumentType get(final String value) {
82          if (value == null) {
83              return JobDocumentType.UNKNOWN;
84          }
85  
86          String val = value.toUpperCase();
87  
88          for (JobDocumentType v : values()) {
89              if (val.equals(v.getType())) {
90                  return v;
91              }
92          }
93  
94          throw new IllegalArgumentException("Unknown document type: '"
95              + value + "'.");
96      }
97  
98  }
