1   package com.ulflander;
2   
3   import com.ulflander.conf.Conf;
4   import com.ulflander.job.Job;
5   import com.ulflander.job.JobFactory;
6   import com.ulflander.model.Document;
7   import com.ulflander.services.JobExecutorService;
8   import com.ulflander.services.Service;
9   import com.ulflander.utils.Log;
10  
11  import java.util.ArrayList;
12  
13  /**
14   * This service is intended only for unit test use.
15   */
16  public class Minethat extends Service {
17  
18      private JobExecutorService service = new JobExecutorService();
19  
20      private ArrayList<String> processorNames;
21  
22      private Boolean verbose = true;
23  
24      public Minethat() {
25          super();
26          if (verbose) {
27              Log.out(Minethat.class, "Minethat starting...");
28          }
29          service.setVerbose(verbose);
30          this.reset();
31      }
32  
33      public void reset() {
34          processorNames = new ArrayList<String>();
35      }
36  
37      public void addProcessor(String s) {
38          if (processorNames.contains(s)) {
39              Log.err(getClass(), "You can't add twice the same processor");
40              return;
41          }
42          processorNames.add(s);
43      }
44  
45      public void removeProcessor(String s) {
46          processorNames.remove(s);
47      }
48  
49      public int getProcessorsSize() {
50          return processorNames != null ? processorNames.size() : 0;
51      }
52  
53      public void submit(String s) {
54          submit(new Document(s));
55      }
56  
57      public void submit(Job j) {
58          service.execute(j);
59      }
60  
61      public void submit(Document d) {
62          Job j = JobFactory.fromDocument(d, processorNames);
63          j.setCustomerId(Conf.getDefaultCID());
64          submit(j);
65      }
66  
67  
68      @Override
69      public String toString() {
70  
71          return "\nWelcome to Minethat\n"
72              + "Minethat is important because it finally allows anybody to "
73              + "do complex computation on documents.\n"
74              + "By offering a framework extensible for document tagging "
75              + "and analysis, it frees users to write specific applications.\n";
76      }
77  }
