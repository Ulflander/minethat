47  
48      public int getSentencesSize() {

50      }
51  
52      public Sentence getSentenceAt(int pos) {
53          return sentences != null && sentences.size() > pos ? sentences.get(pos) : null;
54      }
55  
56      public Token getTokenAt(int sentencePos, int pos) {
57          return getSentenceAt(sentencePos).getTokenAt(pos);
58      }
59  
60      public void appendSentence(Sentence s) {
61          if (s.getRaw().equals("")) {
62              return;
63          }
64  
65          sentences.add(s);
66      }
